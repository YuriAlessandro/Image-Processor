244 
245  /* Cria um único Pixel temporário para servir de auxiliar na função.*/
246  Pixel temp = image[0][0];
247 
248  /* Cria o novo arquivo e escreve as informações básicas da imagem nele.*/
249  FILE *newfile;

251  fprintf(newfile, "%s\n%i %i\n%i\n", filetype, width, height, clrRange);
252 
253  /* Os laços percorrem a matriz procurando por pixels repetidos e escrevendo-os
254  * no arquivo de acordo com o número de vezes que apareceram consecutivamente.*/
255  for(i = 0; i < height; i++){
256  for(j = 0; j < width; j++){

258  mult++;
259  }else{
260  fprintf(newfile, "%i(%i %i %i)\n", mult, temp.red, temp.green, temp.blue);
261  mult = 1;
262  temp = image[i][j];
263  }
